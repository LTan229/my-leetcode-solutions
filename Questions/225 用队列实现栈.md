class MyStack:

    def __init__(self):
        self.q1 = []
        self.q2 = []
        

    def push(self, x: int) -> None:
        self.q1.append(x)
        

    def pop(self) -> int:
        if not self.q1:
            raise Exception
        while self.q1:
            temp = self.q1.pop(0)
            if self.q1:
                self.q2.append(temp)
        while self.q2:
            self.q1.append(self.q2.pop(0))
        return temp
        

    def top(self) -> int:
        temp = self.pop()
        self.q1.append(temp)
        return temp
        

    def empty(self) -> bool:
        return len(self.q1) == 0
