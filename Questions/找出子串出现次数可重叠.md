        lenh = len(haystack)
        lenn = len(needle)
        if lenh < lenn:
            return 0
        
        nxt = [0 for _ in range(lenn)]

        i = 0
        for j in range(1, lenn):
            while i > 0 and needle[i] != needle[j]:
                i = nxt[i-1] # 错：nxt[i] 跳转到已经匹配好字符串的最长前后缀
            if needle[i] == needle[j]:
                i += 1 # 下一个idx，即当前匹配成功长度
            nxt[j] = i # 错：nxt[i] + 1。这个不用放if里，匹配不成功就是0
