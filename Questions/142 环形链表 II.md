# 哈希集合
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None or head.next is None:
            return None
        cur = head
        visited = set([])
        while cur is not None:
            if cur in visited:
                return cur
            else:
                visited.add(cur)
                cur = cur.next
        return None
```

# 快慢指针

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None or head.next is None:
            return None
        fast = head
        slow = head
        while fast is not None and slow is not None:
            try:
                fast = fast.next.next
            except:
                return None
            slow = slow.next
            if fast == slow:
                third = head
                while third != slow:
                    third = third.next
                    slow = slow.next
                return third
        return None
```
