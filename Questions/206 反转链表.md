```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None:
            return head
        rehead, _ = self.subreverse(head)
        return rehead
    
    def subreverse(self, head):
        if head.next is not None:
            rehead, retail = self.subreverse(head.next)
            retail.next = head
            retail = retail.next
            retail.next = None # important!
            return rehead, retail
        else:
            return head, head
```

迭代
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None or head.next is None:
            return head
        prev = None
        cur = head
        while cur is not None:
            next = cur.next
            cur.next = prev
            prev = cur
            cur = next
        return prev
    
```
