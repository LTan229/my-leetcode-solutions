# 双指针
```python
class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        lens = len(s)
        lent = len(t)
        if lens == 0:
            return True
        if lens > lent:
            return False
        i = 0 # s indx
        for j in range(lent):
            if t[j] == s[i]:
                i += 1
                if i == lens:
                    break
        if i < lens:
            return False
        return True
```

# DP
```python
class Solution:
    def isSubsequence(self, s: str, t: str) -> bool:
        lens = len(s)
        lent = len(t)
        if lens == 0:
            return True
        if lens > lent:
            return False
        dp = [[0] * (lent + 1) for _ in range(lens + 1)]
        for i in range(1, lens + 1):
            for j in range(1, lent + 1):
                if s[i - 1] == t[j - 1]:
                    dp[i][j] = dp[i - 1][j - 1] + 1
                else:
                    dp[i][j] = dp[i][j - 1] # s必须都匹配，所以只能退j
        return dp[-1][-1] == lens
```
