class Solution:
    def longestPalindromeSubseq(self, s: str) -> int:
        ls = len(s)
        dp = [[0] * (ls + 1) for _ in range(ls + 1)]
       # 范围内的最长回文子串
       # 相等，对角线
       # 相等，看看内部（甚至还涵盖了长度为2的情况）
       # 不相等，看看内部
        for i in range(1, ls + 1):
            for j in range(i, 0, -1):
                curlen = i - j + 1
                if s[i - 1] == s[j - 1]:
                    if i == j:
                        dp[i][j] = 1
                    else:
                        dp[i][j] = dp[i - 1][j + 1] + 2
                else:
                    dp[i][j] = max(dp[i][j + 1], dp[i - 1][j])

        return dp[-1][1]
