class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        lenums = len(nums)
        if lenums < 3:
            return []
        
        nums.sort() # sort ascending 
        # print(n?ums)
        results = []
        for i in range(lenums): 
            if not (i == 0 or nums[i] != nums[i - 1]):# make sure the item is not selected before
                continue
            j = i + 1
            k = lenums - 1
            while j < k and j < lenums and k > 0:
                # print(i, j, k)
                tsum = nums[i] + nums[j] + nums[k]
                if tsum == 0:
                    results.append([nums[i], nums[j], nums[k]])
                if not tsum < 0:
                    k -= 1
                    while k > j and nums[k] == nums[k + 1]:
                        k -= 1
                if not tsum > 0:
                    j += 1
                    while k > j and nums[j] == nums[j - 1]:
                        j += 1                
        
        return results
