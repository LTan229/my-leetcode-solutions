from collections import deque

nrow, ncol = map(int, input().split())

matrix = []
for _ in range(nrow):
    matrix.append(list(map(int, input().split())))

# search for all isles, mark id and area
isle = dict()
isle_id = 2
directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]

def bfs(i, j, iid):
    q = deque([[i, j]])
    matrix[i][j] = iid
    area = 1
    while q:
        curi, curj = q.popleft()
        for di, dj in directions:
            newi = curi + di
            newj = curj + dj
            if 0 <= newi < nrow and 0 <= newj < ncol and matrix[newi][newj] == 1: # is isle and not visited
                q.append([newi, newj])
                matrix[newi][newj] = iid
                area += 1
    isle[iid] = area

for i in range(nrow):
    for j in range(ncol):
        if matrix[i][j] != 1:
            continue
        bfs(i, j, isle_id)
        isle_id += 1

# print(isle)

if not isle: # no island
    print(1)
else:
    # traverse sea check which isles each point connect and calc area
    maxarea = isle[2]
    for i in range(nrow):
        for j in range(ncol):
            if matrix[i][j] != 0: # for all sea points
                continue
            area = 1 # area of the point
            neighbors = []
            for di, dj in directions:
                newi = i + di
                newj = j + dj
                if 0 <= newi < nrow and 0 <= newj < ncol and matrix[newi][newj] > 1: # only select isles
                    neighbors.append(matrix[newi][newj])
            neighbors = list(set(neighbors))
            for n in neighbors:
                area += isle[n]
            maxarea = max(maxarea, area)

    print(maxarea)
