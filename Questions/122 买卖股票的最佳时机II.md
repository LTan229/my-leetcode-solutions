```python
class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        dp = [- prices[0], 0]
        for i in range(1, len(prices)):
            wval = max(dp[0], dp[1] - prices[i])
            woval = max(dp[1],dp[0] + prices[i])
            dp = [wval, woval]
        return max(dp)
```
第[i]天的本金是上一天（不持）的所有钱，可能包括之前买卖的利润。
