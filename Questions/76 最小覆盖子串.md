class Solution:
    def minWindow(self, s: str, t: str) -> str:
        len_s = len(s)
        len_t = len(t)
        if len_s < len_t: # 一定成立的判定
            return ""
		start, end = 0, 0 # 指针
        t_cnt = Counter(t) 
        s_cnt = Counter(s[0])
        min_len = len_s + 1 # 目前发现的最小长度
        min_window = (start, end) # 目前发现的最小子串
        flag_find = False # 发现过没

        while end < len_s:
            if s_cnt >= t_cnt: # 满足要求
                cur_len = end - start + 1 # 记录
                if min_len > cur_len:
                    flag_find = True
                    min_len = cur_len
                    min_window = (start, end)
                s_cnt[s[start]] -= 1 # 缩小窗口
                if s_cnt[s[start]] == 0: # 检查记录残余
                    s_cnt.pop(s[start])
                start += 1
            else:
                end += 1 # 扩大窗口
                if end < len_s: 
                    s_cnt[s[end]] += 1

        if not flag_find:
            return ""
        return s[min_window[0]:min_window[1]+1]
