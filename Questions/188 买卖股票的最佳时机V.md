class Solution:
    def maxProfit(self, k: int, prices: List[int]) -> int:
        dp = [0] + [- prices[0], 0] * (k)
        for i in range(1, len(prices)):
            for j in range(1, 2 * k + 1, 2):
                w = max(dp[j], dp[j - 1] - prices[i])
                wo = max(dp[j + 1], dp[j] + prices[i])
                dp[j] = w
                dp[j + 1] = wo

        return max(dp)
