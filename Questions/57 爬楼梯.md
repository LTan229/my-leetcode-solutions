```python
# dp[i] = \sum_{j = i - m}^{i - 1} dp[j]
n, m = map(int, input().split())

dp = [1]
for i in range(2, n + 1):
    new = 0
    sum_range = min(len(dp), m)
    for j in range(i - sum_range, i):
        new += dp[j - 1]
    if i <= m:
        new += 1
    dp.append(new)

print(dp[-1])
```
