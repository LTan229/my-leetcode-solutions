# 启发式：发现不匹配从下一个继续找
```python
class Solution:
    def strStr(self, haystack: str, needle: str) -> int:
        lenh = len(haystack)
        lenn = len(needle)
        for hidx in range(lenh - lenn + 1):
            if haystack[hidx] != needle[0]:
                continue
            tidx = hidx
            for nidx in range(lenn):
                if haystack[tidx] != needle[nidx]:
                    break
                tidx += 1
                if nidx == lenn - 1:
                    return hidx
        return -1
```

# KMP
```python
class Solution:
    def strStr(self, haystack: str, needle: str) -> int:
        haystack, needle = list(haystack), list(needle)
        lenh = len(haystack)
        lenn = len(needle)

        # 计算前缀表：截止每个字符（包括）的字符串的最长共同前后缀
        j = -1
        nextarr = [0] * lenn
        nextarr[0] = j
        for i in range(1, lenn):
            while j >= 0 and needle[j + 1] != needle[i]: # mismatch,注意比较对象不同
                j = nextarr[j] # back to the lcp of str[0...j] becuase we need to find a prefix of str[0...j] and a suffix of str[0...i-1]
            if needle[j + 1] == needle[i]: # match
                j += 1 
            nextarr[i] = j # write j++ if match, write -1 if no match，注意目标不同

        # search with next
        j = -1
        for i in range(0, lenh):
            while j >= 0 and needle[j + 1] != haystack[i]: # 出现不匹配的时候进行回退
                j = nextarr[j]
            if needle[j + 1] == haystack[i]: # 如果一直匹配就一直走这个
                j += 1
            if j == lenn - 1: 
                return i - j
        
        return -1
```
