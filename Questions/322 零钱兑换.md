class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        # 处理极端情况
        if amount == 0:
            return 0
        # 求最小值，初始化为一个表示“不可达”或“无限大”的值
        dp = [amount + 1] * (amount + 1)
        dp[0] = 0
        for v in coins:
            for j in range(v, amount + 1):
            # 只要写状态转移函数就可以了，不需要过多判断
                dp[j] = min(dp[j], dp[j - v] + 1)
                # print(dp[j])
        # 处理没找到的情况
        if dp[-1] == amount + 1:
            return -1
        return dp[-1]
