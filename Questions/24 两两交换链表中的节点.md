迭代
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None or head.next is None:
            return head
        cur = head
        empt = ListNode(-1, head)
        prev = empt
        while cur is not None and cur.next is not None:
            next = cur.next.next
            prev.next = cur.next
            prev.next.next = cur
            cur.next = next
            prev = cur
            cur = next
        return empt.next
```
