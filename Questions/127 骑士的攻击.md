```python
import heapq

dirs = [[-1, -2], [-2, -1], [-2, 1], [-1, 2], [1, 2], [2, 1], [2, -1], [1, -2]]

def search(a1, a2, b1, b2):
    def d(i, j):
        return ((i - b1) ** 2 + (j - b2) ** 2) ** (1/2)
    
    def valid(i, j):
        return i > 0 and i < 1000 and j > 0 and j < 1000
        
    goneval = [[float('inf')] * 1000 for _ in range(1000)]

    pq = [(0 + d(a1, a2), a1, a2)]
    goneval[a1][a2] = 0

    while pq:

        fval, i, j = heapq.heappop(pq)

        if i == b1 and j == b2:
            print(goneval[i][j])
            return

        for di, dj in dirs:
            newi, newj = i + di, j + dj
            
            if not valid(newi, newj):
                continue # remember to check validity
    
            if goneval[i][j] + 1 < goneval[newi][newj]: # if the new goneval is less than the old goneval for newi newj
                goneval[newi][newj] = goneval[i][j] + 1
                newf = goneval[newi][newj] + d(newi, newj)
                heapq.heappush(pq, (newf, newi, newj)) # pq selects next node by f


n = int(input())

for _ in range(n):
    a1, a2, b1, b2 = map(int, input().split())
    search(a1, a2, b1, b2)
```
