```python
class Solution:
    def numDistinct(self, s: str, t: str) -> int:
        lens = len(s)
        lent = len(t)
        dp = [[0] * (lent + 1) for _ in range(lens + 1)]
        for i in range(lens + 1):
            dp[i][0] = 1
        for i in range(1, lens + 1):
            for j in range(1, lent + 1):
                if s[i - 1] == t[j - 1]:
                    dp[i][j] = dp[i - 1][j] + dp[i - 1][j - 1]
                else:
                    dp[i][j] = dp[i - 1][j]
        # print(dp)
        return dp[-1][-1]         
```
