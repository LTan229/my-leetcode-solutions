很复杂的做法
```python
from collections import deque
 
nrow, ncol = map(int, input().split())
matrix = []
for i in range(nrow):
    matrix.append(list(map(int, input().split())))
 
directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]
for i in range(nrow):
    for j in range(ncol):
        if matrix[i][j] == 0:
            continue
 
        curarea = 1
        isisle = i * j  * (i - (nrow - 1)) * (j - (ncol - 1))!= 0
 
        visited = [[0] * ncol for _ in range(nrow)]
        visited[i][j] = 1
        q = deque([[i, j]])
        while q:
            curi, curj = q.popleft()
            for di, dj in directions:
                newi = curi + di
                newj = curj + dj
                if (0 <= newi < nrow and 0 <= newj < ncol and # valid
                    matrix[newi][newj] == 1 and # is land
                    visited[newi][newj] == 0): # not visited
                    if newi * newj * (newi - (nrow - 1)) * (newj - (ncol - 1)) == 0:
                        isisle = False
                    curarea += 1
                    visited[newi][newj] = 1
                    q.append([newi, newj])
        # print(visited, isisle)
        if isisle:
            for curi in range(nrow):
                for curj in range(ncol):
                    if visited[curi][curj] == 1:
                        matrix[curi][curj] = 0
 
for row in matrix:
    print(*row)
```

搜索边界标记大陆然后沉没孤岛
```python
from collections import deque

nrow, ncol = map(int, input().split())
matrix = []
for i in range(nrow):
    matrix.append(list(map(int, input().split())))

q = deque([])
for row in range(nrow):
    for i, j in [[row, 0], [row, ncol - 1]]:
        if matrix[i][j] == 1:
            q.append([i, j])
for col in range(ncol):
    for i, j in [[0, col], [nrow - 1, col]]:
        if matrix[i][j] == 1:
            q.append([i, j])

directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]

while q:
    i, j = q.popleft()
    matrix[i][j] = 2
    for di, dj in directions:
        newi = i + di
        newj = j + dj
        if -1 < newi < nrow and -1 < newj < ncol and matrix[newi][newj] == 1:
            matrix[newi][newj] = 2
            q.append([newi, newj])

for i in range(nrow):
    for j in range(ncol):
        if matrix[i][j] > 0:
            matrix[i][j] -= 1

for row in matrix:
    print(*row)
```
