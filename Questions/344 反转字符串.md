# 双指针
```python
class Solution:
    def reverseString(self, s: List[str]) -> None:
        """
        Do not return anything, modify s in-place instead.
        """
        lens = len(s)
        left = 0
        right = lens - 1
        while left < right:
            t = s[left]
            s[left] = s[right]
            s[right] = t
            left += 1
            right -= 1
        
```

# 分解为旋转替换操作
原地进行
- 先完整旋转
- 然后分别旋转两个子部分
```python
k = int(input())
s = input()
sarr = list(s)

# inverse whole
lenstr = len(s)
left = 0
right = lenstr - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1

# inverse first k
left = 0
right = k - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1

# inverse rear
left = k
right = lenstr - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1

print("".join(sarr))
```
