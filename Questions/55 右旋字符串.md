# 字符串操作
```python
k = int(input())
istr = input()

ostr = istr[-k:] + istr[:-k]
print(ostr)
```

# 反转
```python
k = int(input())
s = input()
sarr = list(s)
 
# inverse whole
lenstr = len(s)
left = 0
right = lenstr - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1
 
# inverse first k
left = 0
right = k - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1
 
# inverse rear
left = k
right = lenstr - 1
while left < right:
    t = sarr[left]
    sarr[left] = sarr[right]
    sarr[right] = t
    left += 1
    right -= 1
 
print("".join(sarr))
```
