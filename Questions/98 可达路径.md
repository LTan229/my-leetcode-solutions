```python
n, m = map(int, input().split())
adjarr = [[] for _ in range(n + 1)]
for _ in range(m):
    s, t = map(int, input().split())
    adjarr[s].append(t)
# for i in range(n):
    # adjarr[i] = sorted(adjarr[i])

results = []
temp = [1]

def dfs(k):
    if k == n:
        results.append(temp[:])
        return
    for i in adjarr[k]:
        temp.append(i)
        dfs(i)
        temp.pop()

dfs(1)

if results:
    for r in results:
        print(*r) # 星号解包
else:
    print(-1)
```
