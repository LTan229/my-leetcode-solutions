class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        nums2idx = {n: idx for idx, n in enumerate(nums2)} # map numbers in nums2 to its idx
        ng = [-1] * len(nums2)
        stack = []
        for i, n in enumerate(nums2):
            # for each item, pop all smaller itemsbefore it
            while stack and n > nums2[stack[-1]]:
                ng[stack[-1]] = nums2[i] # record next bigger item 
                stack.pop()
            stack.append(i) # push the current item into the stack 
        results = []
        for n in nums1:
            results.append(ng[nums2idx[n]]) # collect results for items in nums1
        return results
