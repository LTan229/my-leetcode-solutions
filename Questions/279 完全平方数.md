class Solution:
    def numSquares(self, n: int) -> int:
        dp = [n] * (n + 1)
        dp[0] = 0
        for i in range(101):
            sq = i * i
            if sq > n:
                break
            for j in range(sq, n + 1):
                dp[j] = min(dp[j], dp[j - sq] + 1)
        # print(dp)
        return dp[-1]
