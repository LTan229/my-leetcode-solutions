```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rob(self, root: Optional[TreeNode]) -> int:
        wroot, woroot = self.robnode(root)
        return max(wroot, woroot)
    
    def robnode(self, root):
        if not root:
            return 0, 0
        # if not (root.left or root.right): # no child
        #     return root.val, 0
        leftw, leftwo = self.robnode(root.left)
        rightw, rightwo = self.robnode(root.right)
        woroot = max(leftw, leftwo) + max(rightw, rightwo)
        wroot = root.val + leftwo + rightwo

        return wroot, woroot
```
