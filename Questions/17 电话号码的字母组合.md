```python
from collections import deque

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        dic = {
            '2':['a', 'b', 'c'],
            '3':['d', 'e', 'f'],
            '4':['g','h','i'],
            '5':['j','k','l'],
            '6':['m','n','o'],
            '7':['p','q','r', 's'],
            '8':['t','u','v'],
            '9':['w','x','y', 'z']
        }

        digits = list(digits)
        results = deque()
        for d in digits:
            if not results:
                for alph in dic[d]:
                    results.append(alph)
                continue
            lenr = len(results)
            for i in range(lenr):
                item = results.popleft()
                for alph in dic[d]:
                    results.append(item + alph)
        return list(results)
```
