```python
class Solution:
    def trap(self, height: List[int]) -> int:
        lenh = len(height)
        leftmax = [-1] * lenh
        lmax, rmax = -1, -1
        for i in range(lenh):
            lmax = max(lmax, height[i])
            leftmax[i] = lmax
        results = 0
        for i in range(lenh - 1, -1, -1):
            rmax = max(rmax, height[i])
            results += min(leftmax[i], rmax) - height[i]
        return results
```

# 双指针
如果另一边不低于当前位置，则当前位置必定是瓶颈
```python
class Solution:
    def trap(self, height: List[int]) -> int:
        lenh = len(height)
        left, right = 0, lenh - 1
        lmax, rmax = 0, 0
        ans = 0
        while left < right: # while not all considered
            if height[left] < height[right]: # left is sure to be bottleneck
                if height[left] < lmax: # prviously has number larger than left
                    ans += lmax - height[left] # store water 
                else: # previously no number larger than left
                    lmax = height[left] # cannot store water
                left += 1 # move pointer forward
            else:
                if height[right] < rmax:
                    ans += rmax - height[right]
                else:
                    rmax = height[right]
                right -= 1
        return ans
```

# 单调栈
```python
class Solution:
    def trap(self, height: List[int]) -> int:
        if len(height) < 3:
            return 0

        stack = [0]
        ans = 0
        for i in range(1, len(height)):
            if height[i] < height[stack[-1]]:
                stack.append(i)
            elif height[i] == height[stack[-1]]:
                stack.pop()
                stack.append(i) # replace prev equal bin to last. this is a kind of merge. can be omitted
            else: # 弹一个算一个桶，这样宽度高度都是确定的
                while stack and height[i] >= height[stack[-1]]:
                    mid = stack.pop()
                    if stack:
                        leftidx = stack[-1]
                        level = min(height[i], height[leftidx])
                        width = i - leftidx - 1
                        ans += width * (level - height[mid])
                stack.append(i)
        return ans
```
