```python
class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        m, n = len(obstacleGrid), len(obstacleGrid[0])

        if obstacleGrid[m - 1].[n - 1] == 1 or obstacleGrid[0].[0] == 1:
            return 0

        obstacleGrid[m - 1].[n - 1] = 1
        
        i, j = m - 1, n - 1
        while i + j > 0:
            if j == 0:
                i -= 1
                j = n - 1
            else:
                j -= 1
            if obstacleGrid[i].[j] == 1:
                obstacleGrid[i].[j] = -1
                continue
            dcnt = obstacleGrid[i + 1].[j] if i < m - 1 and obstacleGrid[i + 1].[j] != -1 else 0
            rcnt = obstacleGrid[i].[j + 1] if j < n - 1 and obstacleGrid[i].[j + 1] != -1 else 0
            obstacleGrid[i].[j] = dcnt + rcnt
        
        return obstacleGrid[0].[0]
```
