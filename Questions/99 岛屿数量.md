DFS会爆栈
```python
from collections import deque

nrow, ncol = map(int, input().split())
matrix = []
for i in range(nrow):
    matrix.append(list(map(int, input().split())))

islecnt = 0
directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]
for i in range(nrow):
    for j in range(ncol):
        if matrix[i][j] < 1:
            continue

        islecnt += 1
        matrix[i][j] = -1

        q = deque([[i, j]])
        while q:
            curi, curj = q.popleft()
            for di, dj in directions:
                newi = curi + di
                newj = curj + dj
                if 0 <= newi < nrow and 0 <= newj < ncol and matrix[newi][newj] == 1:
                    matrix[newi][newj] = -1 # check at add not pop
                    q.append([newi, newj])
                    # print(curi, curj, newi, newj)

print(islecnt)
```
