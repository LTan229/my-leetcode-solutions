class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        sums = sum(nums)
        if sums % 2 != 0:
            return False
        target = sums // 2
        bag = [0] * (target + 1)
        for n in nums:
            for w in range(target, -1, -1):
                if w >= n:
                    bag[w] = max(n + bag[w - n], bag[w])
        return bag[-1] == target
