class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        
        return self.dfs(9, k, n)

    def dfs(self, maxval, k, n):
        if maxval > n:
            maxval = n
        if maxval < 1:
            return []
        if n == maxval and k == 1:
            return [[maxval]]
        # select maxval
        s_results = self.dfs(maxval - 1, k - 1, n - maxval)
        if s_results:
            for i in range(len(s_results)):
                s_results[i] += [maxval]
        # not select maxval
        ns_results = self.dfs(maxval - 1, k, n)
        return s_results + ns_results
