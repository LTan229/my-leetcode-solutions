# 原生
```python
ncity, nroad = map(int, input().split())

edges = []
for _ in range(nroad):
    edges.append(tuple(map(int, input().split())))

distances = [float('inf') for _ in range(ncity + 1)]
distances[1] = 0 # 不要忘记初始化

for round in range(ncity - 1):
    for s, t, v in edges:
        if distances[s] + v < distances[t]:
            distances[t] = distances[s] + v

if distances[-1] == float('inf'):
    print("unconnected")
else:
    print(distances[-1])
```

# 队列加速

```python
from collections import deque

ncity, nroad = map(int, input().split())

edges = [[] for _ in range(ncity + 1)]
for _ in range(nroad):
    s, t, v = map(int, input().split())
    edges[s].append((t, v))

distances = [float('inf') for _ in range(ncity + 1)]
distances[1] = 0
inq = [False for _ in range(ncity + 1)]
q = deque([1])
inq[1] = True

while q:
    node = q.popleft()
    inq[node] = False
    for neib, dist in edges[node]:
        if distances[neib] > distances[node] + dist:
            distances[neib] = distances[node] + dist
            if not inq[neib]: # only add neib if it is not in queue already, it will be updated if it is in queue anyway
	            q.append(neib)
	            inq[neib] = True

if distances[-1] == float('inf'):
    print("unconnected")
else:
    print(distances[-1])
```
