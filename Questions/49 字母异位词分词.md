from collections import defaultdict
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = defaultdict(list) # eliminate check for key existence
        orda = ord('a')
        for s in strs:
            cnt = [0] * 26 # bincount as identifier
            for c in s:
                cnt[ord(c) - orda] += 1
            d[tuple(cnt)].append(s) # un-changeable tuple as key
        return list(d.values())
