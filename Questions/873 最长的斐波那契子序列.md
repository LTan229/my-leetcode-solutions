import java.util.*;

class Solution {
    public int lenLongestFibSubseq(int[] arr) {
        // Scanner in = new Scanner(System.in);
        // if (!in.hasNext()) return;
        
        // // 1. 处理输入（这是最烦人的部分）
        // String str = in.next();
        // // 顺便处理一下可能存在的方括号 [1,2,3]
        // str = str.replace("[", "").replace("]", "");
        // String[] parts = str.split(",");
        // int n = parts.length;
        // int[] arr = new int[n];
        // Map<Integer, Integer> indexMap = new HashMap<>();
        // for (int i = 0; i < n; i++) {
        //     arr[i] = Integer.parseInt(parts[i].trim());
        //     indexMap.put(arr[i], i); // 把值和索引存起来，方便快速找到 arr[i]
        // }

        Map<Integer, Integer> indexMap = new HashMap<>();
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            indexMap.put(arr[i], i);
        }

        // 2. 动态规划
        // dp[j][k] 表示以 index j 和 index k 结尾的序列长度
        // 要确定下一个数是不是斐波那契数列，你要知道前面两个数
        int[][] dp = new int[n][n];
        // 如果要初始化就初始化所有dp[j][k] = 2 （因为任意两个数都可以看作长度为 2 的斐波那契起点）
        int maxLen = 0;

        // 遍历所有可能的最后两个数 arr[j] 和 arr[k]
        // 因为dpjk依赖dpij，i<j<k，所以不管顺序从小到大遍历jk就可以
        for (int k = 0; k < n; k++) {
            for (int j = 0; j < k; j++) {
                // 我们想找 arr[i] 使得 arr[i] + arr[j] = arr[k]
                // 即 arr[i] = arr[k] - arr[j]
                int target = arr[k] - arr[j];
                
                // 斐波那契序列性质：arr[i] < arr[j] < arr[k]
                if (target < arr[j] && indexMap.containsKey(target)) {
                    int i = indexMap.get(target);
                    // 如果存在 i，则在 dp[i][j] 的基础上加 1
                    // 如果dpij是0, 说明ijk第一次凑齐三个数，初始长度是 3 如果dpij>=3,说明在一个现有的数列上增加了一个数
                    dp[j][k] = Math.max(dp[i][j] + 1, 3);
                    maxLen = Math.max(maxLen, dp[j][k]);
                }
            }
        }
        // 斐波那契数列的定义是长度必须>=3
        // System.out.print(maxLen >= 3 ? maxLen : 0);
        return maxLen >= 3 ? maxLen : 0;
    }
}
