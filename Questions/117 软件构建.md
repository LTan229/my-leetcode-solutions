```python
import heapq

nnode, nedge = map(int, input().split())

edges = [[] for _ in range(nnode)]
indegree = [0] * nnode
for _ in range(nedge):
    s, t = map(int, input().split())
    edges[s].append(t)
    indegree[t] += 1

pq = [] # only contain the nodes with 0 in-deg, do not put all of them in.
for i in range(nnode):
    if indegree[i] == 0:
        heapq.heappush(pq, i)

results = []
while pq:
    cur = heapq.heappop(pq)
    results.append(cur)

    for e in edges[cur]:
        indegree[e] -= 1
        if indegree[e] == 0:
            heapq.heappush(pq, e)

if len(results) < nnode:
    print(-1)
else:
    print(*results)
```
