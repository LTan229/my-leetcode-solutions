class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        #for each queen, iterate the next emply row so that each valid position is tried
        self.results = []
        self.temp = []
        self.n = n
        self.dfs(n, 0)

        return self.results
    
    def dfs(self, n, cur):
        if cur == n:
            temp = []
            for row in range(n):
                temp.append("." * self.temp[row] + "Q" + "." * (n - self.temp[row] - 1))
            self.results.append(temp)
            return
        validpos = self.getvalidpos()
        for col in validpos:
            self.temp.append(col)
            self.dfs(n, cur + 1)
            self.temp.pop()
        return
    
    def getvalidpos(self):
        # for each existing queen
        # get its col val and its diagonal val on next row if they are in range 0, n
        validpos = set(range(self.n))
        currow = len(self.temp)
        for row, col in enumerate(self.temp):
            validpos.discard(col)
            distance = currow - row
            left, right = col - distance,  col + distance
            validpos.discard(left)
            validpos.discard(right)
        return sorted(list(validpos))
