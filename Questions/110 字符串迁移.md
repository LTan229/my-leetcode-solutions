from collections import deque

# read words
n = int(input())
beginstr, endstr = input().split()
strlist = [endstr]
for _ in range(n):
    strlist.append(input())

def distance(str1, str2):
    cnt = 0
    for i in range(len(str1)):
        if str1[i] != str2[i]:
            cnt += 1
    return cnt

visited = [False] * len(strlist)
q = deque([(beginstr, 1)])
while q:
    curstr, curstep = q.popleft()
    if curstr == endstr:
        print(curstep)
        exit()
    for i, nextstr in enumerate(strlist):
        if not visited[i] and distance(curstr, nextstr) == 1:
            q.append((nextstr, curstep + 1))
            visited[i] = True
            # print(nextstr, curstep + 1)

# 5. 字母比较相同的时候距离才++
