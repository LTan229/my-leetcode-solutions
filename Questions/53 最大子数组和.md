```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        lennums = len(nums)
        if lennums == 1:
            return nums[0]
        dp = [0] # usecur, nocur
        maxx = - 1e5
        for i in range(1, lennums + 1):
            # 不用当前：前面的和全部归零
            # nocurt = 0
            # 用当前：前面有/前面无
            usecurt = max(dp[-1], 0) + nums[i - 1]
            dp.append(usecurt)
            maxx = max(maxx, usecurt)
        print(dp)
        return maxx
```
