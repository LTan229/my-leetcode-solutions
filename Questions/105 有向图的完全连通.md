from collections import deque

n, k = map(int, input().split())
adjlist = [[] for _ in range(n + 1)]
for _ in range(k):
    s, t = map(int, input().split())
    adjlist[s].append(t)

visited = [0] * (n + 1)

q = deque([1])
visited[1] = 1
while q:
    cur = q.popleft()
    for nextnode in adjlist[cur]:
        if visited[nextnode] == 0:
            q.append(nextnode)
            visited[nextnode] = 1

if sum(visited) == n:
    print(1)
else:
    print(-1)
