class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        self.temp = []
        self.result = []
        self.candidates = sorted(candidates)
        self.dfs(0, target)
        return self.result
    
    def dfs(self, cur, target):
        if target == 0:
            self.result.append(self.temp[:])
            return None
        
        if cur == len(self.candidates) or target < self.candidates[cur]:
            return None 
        
        self.dfs(cur + 1, target)

        self.temp.append(self.candidates[cur])
        self.dfs(cur, target - self.candidates[cur])
        self.temp.pop()
