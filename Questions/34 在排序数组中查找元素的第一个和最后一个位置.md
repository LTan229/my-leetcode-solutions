class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        len_num = len(nums)
        left = 0
        right = len_num - 1

        while right >= left:
            mid = left + (right - left) // 2
            if nums[mid] == target:
                left_ans = self.splitSeatchSmall(nums, target, 0, mid)
                right_ans = self.splitSeatchBig(nums, target, mid, len_num - 1)
                return [left_ans,right_ans]
            elif nums[mid] > target:
                right = mid - 1
            else:
                left = mid + 1
            
        return [-1,-1]

    def splitSeatchSmall(self, nums, target, left, right):
        while right >= left:
            mid = left + (right - left) // 2
            if nums[mid] == target:
                right = mid - 1
            # elif nums[mid] > target:
            #     right = mid - 1
            else:
                left = mid + 1
        return left

    def splitSeatchBig(self, nums, target, left, right):
        while right >= left:
            mid = left + (right - left) // 2
            if nums[mid] == target:
                left = mid + 1
            else: # nums[mid] > target
                right = mid - 1
            # else:
            #     left = mid + 1
        return right
