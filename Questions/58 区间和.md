```python
n = int(input())
arr = []
sums = []
cur_sum = 0
for i in range(n):
    arr.append(int(input()))
    cur_sum += arr[i]
    sums.append(cur_sum)

while True:
    try:
        start, end = input().split(' ')
        start = int(start)
        end = int(end)
        if start == 0:
            print(sums[end])
        else:
            print(sums[end] - sums[start - 1])
    except:
        break
    
```
