```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        # handle edge and leaf case
        nodecnt = len(inorder)
        if nodecnt == 0:
            return None
        # get root from the last of post order
        rootval = postorder[-1]
        # use root to partition in order into left and right
        rtidx = inorder.index(rootval)
        # for each group, get root from the last of post order
        leftroot = self.buildTree(inorder[0:rtidx], postorder[0:rtidx])
        rightroot = self.buildTree(inorder[rtidx + 1:nodecnt], postorder[rtidx: nodecnt - 1])

        root = TreeNode(rootval)
        root.left = leftroot
        root.right = rightroot
        
        return root
```
