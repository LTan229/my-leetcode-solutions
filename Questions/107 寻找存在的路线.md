```python
nnode, nedge = map(int, input().split())

parents = [i for i in range(nnode + 1)]

def find(u):
    if parents[u] == u:
        return u
    else:
        parents[u] = find(parents[u])
        return parents[u]

def join(s, t): # s->t
    s = find(s)
    t = find(t)
    if s != t:
        parents[t] = s

for _ in range(nedge):
    s, t = map(int, input().split())
    join(s, t)

src, dest = map(int, input().split())

if find(src) == find(dest):
    print(1)
else:
    print(0)

# nnode直接按照题目定义写成了n
# for _ in nedge
# parents[s] = t写成了map[s] = t
```
