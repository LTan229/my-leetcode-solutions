class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        stack = []
        lennums = len(nums)
        results = [-1] * lennums
        maxx = max(nums)
        for idx in range(2 * lennums):
            i = idx % lennums
            while stack and nums[i] > nums[stack[-1]]:
                results[stack[-1]] = nums[i]
                stack.pop()
            if idx < lennums:
	            stack.append(i)
            if idx > lennums and nums[i] == maxx:
                break
        return results
