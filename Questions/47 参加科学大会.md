# 普通做法
```python
nnode, nedge = map(int, input().split())

adjlist = [[] for _ in range(nnode + 1)]
for _ in range(nedge):
    s, e, v = map(int, input().split())
    adjlist[s].append((e, v))

distance = [float('inf')] * (nnode + 1) # stores the min total distance from node 1 to each point

cur = 1
distance[cur] = 0
visited = [False] * (nnode + 1)

for _ in range(nnode): # 循环n次即可
    visited[cur] = True
    
    # update nearest point
    for neib, time in adjlist[cur]:
        distance[neib] = min(distance[neib], distance[cur] + time)
        
    # get next not visited with minimal distance
    mintime = float('inf')
    for node in range(1, nnode + 1):
        if not visited[node] and distance[node] < mintime:
            mintime = distance[node]
            cur = node

if distance[-1] == float('inf'):
    print(-1)
else:
    print(distance[-1])
```

# 最小堆

懒惰删除：当发现一条更短的路径时，**不要去堆里找旧数据并修改它**，而是直接**把新的更短的距离压入堆中**。在最短数据访问后标记该节点已经访问

```python
import heapq

nnode, nedge = map(int, input().split())

adjlist = [[] for _ in range(nnode + 1)]
for _ in range(nedge):
    s, e, v = map(int, input().split())
    adjlist[s].append((e, v))

distance = [float('inf')] * (nnode + 1)
visited = [False] * (nnode + 1)

distance[1] = 0
pq = [(0, 1)]

while pq:
    dist, cur = heapq.heappop(pq)

    if visited[cur]:
        continue

    visited[cur] = True

    if cur == nnode:
        print(dist)
        exit()

    for neib, nextdist in adjlist[cur]:
        newdist = dist + nextdist
        if newdist < distance[neib] and not visited[neib]:
            distance[neib] = newdist
            heapq.heappush(pq, (dist + nextdist, neib))
    
print(-1)

```
