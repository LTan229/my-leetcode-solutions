class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        matrix = [[0] * n for _ in range(n)]
        row, col, dir_i = 0, 0, 0
        for i in range(n * n):
            matrix[row][col] = i + 1
            dr, dc = dirs[dir_i]
            r, c = row + dr, col + dc
            if r < 0 or r >= n or c < 0 or c >= n or matrix[r][c] > 0:
                dirIdx = (dir_i + 1) % 4 
                dr, dc = dirs[dirIdx]
            row, col = row + dr, col + dc
        
        return matrix
