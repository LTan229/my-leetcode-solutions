```python
n = int(input())

parent = [i for i in range(n + 1)]

def find(s):
    if s == parent[s]:
        return s
    else:
        parent[s] = find(parent[s])
        return parent[s]

for _ in range(n):
    s, t = map(int, input().split())
    ps = find(s)
    pt = find(t)
    if ps == pt:
        print(s, t)
        exit()
    else:
        parent[pt] = ps

# when the father of both nodes are the same, it means i the edge is creating circle
```
