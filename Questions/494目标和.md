```python
class Solution:
    def findTargetSumWays(self, nums: List[int], target: int) -> int:
        sums = sum(nums)
        if (sums - target) % 2 == 1:
            return 0
        if abs(target) > sums:
            return 0
        bagsize = (sums - target) // 2
        bag = [0] * (bagsize + 1)
        bag[0] = 1
        for num in nums:
            for j in range(bagsize, num - 1, -1):
                    bag[j] = bag[j] + bag[j - num]
        print(bag)
        return bag[-1]
```
