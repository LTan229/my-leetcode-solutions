```python
class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        lens = len(s)
        
        # 优化：将字典列表转换为哈希集合，查找时间复杂度从 O(N) 降至 O(1)
        word_set = set(wordDict)
        
        # dp[i] 表示字符串 s 的长度为 i 的前缀 s[0...i-1] 是否可以被拆分
        # 长度为 lens + 1
        dp = [False] * (lens + 1)
        
        # dp[0] 为 True，表示空字符串可以被拆分（作为递归的起点）
        dp[0] = True 
        
        # 外层循环：i 从 1 到 lens，遍历所有前缀的长度
        for i in range(1, lens + 1):
            # 内层循环：j 从 0 到 i-1，遍历所有可能的切分点
            # j 表示切分后的前缀长度
            for j in range(i):
                # j 是切分点：
                # 1. 前缀 s[0...j-1] (长度为 j) 必须可拆分 (dp[j] == True)
                # 2. 子串 s[j...i-1] (长度为 i-j) 必须在字典中
                
                current_word = s[j:i]
                
                if dp[j] and current_word in word_set:
                    dp[i] = True
                    # 一旦找到一个成功的拆分方式，就可以跳出内层循环
                    break 
        
        # 最终结果：dp[lens] 表示整个字符串 s 是否可被拆分
        return dp[lens]
```
