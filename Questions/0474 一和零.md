class Solution:
    def findMaxForm(self, strs: List[str], m: int, n: int) -> int:
        dp = [[0] * (n + 1) for _ in range(m + 1)] # max size of subset with m 0s and n 1s

        for sid, s in enumerate(strs):
            cnt0 = s.count('0')
            cnt1 = s.count('1')
            for i in range(m, cnt0 - 1, -1): # 0s
                for j in range(n, cnt1 - 1, -1): # 1s
                    dp[i][j] = max(dp[i][j], dp[i - cnt0][j - cnt1] + 1)

        return dp[-1][-1]
