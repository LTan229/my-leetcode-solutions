```python
from collections import deque

nrow, ncol = map(int, input().split())
matrix = []
for i in range(nrow):
    matrix.append(list(map(int, input().split())))

sumarea = 0
directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]
for i in range(nrow):
    for j in range(ncol):
        if matrix[i][j] < 1:
            continue

        curarea = 1
        isisle = i * j  * (i - (nrow - 1)) * (j - (ncol - 1)) != 0
        matrix[i][j] = -1

        q = deque([[i, j]])
        while q:
            curi, curj = q.popleft()
            for di, dj in directions:
                newi = curi + di
                newj = curj + dj
                if 0 <= newi < nrow and 0 <= newj < ncol and matrix[newi][newj] == 1:
                    if newi * newj * (newi - (nrow - 1)) * (newj - (ncol - 1)) == 0:
                        isisle = False
                    curarea += 1
                    matrix[newi][newj] = -1 # check at add not pop

                    q.append([newi, newj])
        if isisle:
            sumarea += curarea

print(sumarea)
```

```python
from collections import deque

nrow, ncol = map(int, input().split())
matrix = []
for i in range(nrow):
    matrix.append(list(map(int, input().split())))

edges = []
for row in range(nrow):
    edges += [[row, 0], [row, ncol - 1]]
for col in range(ncol):
    edges += [[0, col], [nrow - 1, col]]

directions = [[1, 0], [-1, 0], [0, 1], [0, -1]]

for i, j in edges:
    if matrix[i][j] == 0:
        continue
    
    matrix[i][j] = 0
    q = deque([[i, j]])
    while q:
        curi, curj = q.popleft()
        for di, dj in directions:
            newi = curi + di
            newj = curj + dj
            if -1 < newi < nrow and -1 < newj < ncol and matrix[newi][newj] == 1:
                matrix[newi][newj] = 0
                q.append([newi, newj])

print(sum([sum(row) for row in matrix]))
```
