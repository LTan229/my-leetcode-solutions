class Solution:
    def totalFruit(self, fruits: List[int]) -> int:
        left, right = 0, 0
        fruit_len = len(fruits)
        maxlen = -1
        cnt = Counter()
        cnt[fruits[0]] += 1

        while right < fruit_len:
            curtype = len(cnt)
            if curtype <= 2:  
                curlen = right - left + 1
                maxlen = curlen if curlen > maxlen else maxlen
                
                right += 1
                if right >= fruit_len:
                    continue
                cnt[fruits[right]] += 1
            else:
                cnt[fruits[left]] -= 1
                if cnt[fruits[left]] == 0:
                    cnt.pop(fruits[left])
                left += 1
        return maxlen
