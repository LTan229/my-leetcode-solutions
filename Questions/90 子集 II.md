from collections import Counter 
class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        self.numcntr = Counter(nums).most_common()
        self.results = [[]]
        self.temp = []
        self.lencnt = len(self.numcntr)
        self.dfs(0)
        return self.results
    
    def dfs(self, cur):
        if cur == self.lencnt:
            return
        
        # no cur num added
        self.dfs(cur + 1)

        # cur num added
        for i in range(self.numcntr[cur][1]):
            self.temp.append(self.numcntr[cur][0])
            self.results.append(self.temp[:])
            self.dfs(cur + 1)
        for i in range(self.numcntr[cur][1]):
            self.temp.pop()
        
        return
