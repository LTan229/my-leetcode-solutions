```python
class MyQueue:

    def __init__(self):
        self.in_stack = []
        self.out_stack = []       
        

    def push(self, x: int) -> None:
        self.in_stack.append(x)

    def pop(self) -> int:
        if len(self.out_stack) is 0:
            lenin = len(self.in_stack)
            while lenin > 0:
                item = self.in_stack.pop()
                self.out_stack.append(item)
                lenin -= 1

        return self.out_stack.pop()   

    def peek(self) -> int:
        item = self.pop()
        self.out_stack.append(item)
        return item

    def empty(self) -> bool:
        return len(self.in_stack) == 0 and len(self.out_stack) == 0
```
