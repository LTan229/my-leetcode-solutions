```python
class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        self.results = [[]]
        self.temp = []
        self.nums = nums
        self.lens = len(nums)
        self.dfs(0)
        return self.results
    
    def dfs(self, cur):
        if cur == self.lens:
            # self.results.append(self.temp[:])
            return
        # do not add cur    
        self.dfs(cur + 1)

        self.temp.append(self.nums[cur])
        self.results.append(self.temp[:])
        self.dfs(cur + 1)
        self.temp.pop()
```
