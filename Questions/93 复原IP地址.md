class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        if len(s) < 4:
            return []
        if len(s) == 4:
            return [".".join(list(s))]
        
        self.temp = []
        self.results = []
        self.dfs(s, 3)
        return self.results
    
    def dfs(self, s, scnt):
        if not s:
            return
        if scnt == 0:
            if int(s) > 255:
                return
            if s.startswith("0") and len(s) > 1:
                return 
            self.temp.append(s)
            self.results.append(".".join(self.temp[:]))
            print("Added", self.temp)
            self.temp.pop()
            return

        maxx = min(len(s), 3)
        for i in range(1, maxx + 1):
            ts = s[:i]
            print(ts)
            if int(ts) > 255:
                continue
            if ts.startswith("0") and len(ts) > 1:
                continue
            self.temp.append(ts)
            print(self.temp)
            self.dfs(s[i:], scnt - 1)
            self.temp.pop()
        return
