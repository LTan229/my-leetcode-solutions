```SQL
with class as (
    select 'Low Salary' as cate union all 
    select 'Average Salary' union all 
    select 'High Salary' 
),
cates as (
    select account_id, CASE
        when income < 20000 then 'Low Salary'
        when income > 50000 then 'High Salary'
        else 'Average Salary'
        END as cate
    from Accounts
)
select class.cate as category, COUNT(account_id) as accounts_count 
from cates right join class on cates.cate = class.cate
group by class.cate

```
