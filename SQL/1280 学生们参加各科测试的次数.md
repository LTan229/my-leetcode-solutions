```SQL
# Write your MySQL query statement below 
select 
	StuSub.student_id, 
	StuSub.student_name, 
	StuSub.subject_name, 
	IFNULL(StuExam.attended_exams, 0) as attended_exams 
from (
	select 
		Students.student_id, 
		Students.student_name, 
		Examinations.subject_name, 
		COUNT(*) as attended_exams 
	from Students 
	left join Examinations 
	on Students.student_id = Examinations.student_id 
	group by 
		student_id, 
		subject_name 
) as StuExam 
right join (
select 
	Students.student_id, 
	Students.student_name, 
	Subjects.subject_name 
from Students 
cross join subjects
) as StuSub 
on StuExam.student_id = StuSub.student_id 
and StuExam.subject_name = StuSub.subject_name 
order by StuSub.student_id, StuSub.subject_name
```
