```SQL
select s.id, CASE
    when s.id % 2 = 1 and even.student is null then s.student
    when s.id % 2 = 1 then even.student
    else odd.student 
END as student
-- select *
from Seat as s 
left join Seat as odd on s.id - 1 = odd.id
left join Seat as even on s.id + 1 = even.id

```
