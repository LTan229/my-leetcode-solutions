```SQL
# Write your MySQL query statement below

select ROUND(SUM(next_day.nd is not null) / COUNT(DISTINCT Activity.player_id), 2) as fraction
from Activity
left join (select player_id, DATE_ADD(MIN(event_date), INTERVAL 1 DAY) as nd from Activity group by player_id) as next_day
on Activity.player_id = next_day.player_id and next_day.nd = Activity.event_date

```
