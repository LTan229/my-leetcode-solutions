```SQL
# Write your MySQL query statement below

select ROUND(SUM(CASE when order_date = customer_pref_delivery_date then 1 else 0 END) * 100 / COUNT(*), 2) as immediate_percentage 
from
(select customer_id, MIN(order_date) as first_date from Delivery group by customer_id) as first_order
join Delivery on first_order.customer_id = Delivery.customer_id and Delivery.order_date = first_order.first_date

```
