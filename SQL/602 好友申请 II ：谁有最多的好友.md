```SQL
with ofcnt as (
    select requester_id as ouid, COUNT(*) as ocnt
    from RequestAccepted 
    group by requester_id
),
ifcnt as (
    select accepter_id as iuid, COUNT(*) as icnt
    from RequestAccepted
    group by accepter_id    
),
jointcnt as(
    select * from ofcnt left join ifcnt on ofcnt.ouid = ifcnt.iuid
    union
    select * from ofcnt right join ifcnt on ofcnt.ouid = ifcnt.iuid
),
sumcnt as (
    select CASE
    when ouid is null then iuid else ouid
    END as uid, IFNULL(ocnt, 0) + IFNULL(icnt, 0) as total
    from jointcnt
)
select uid as id, total as num
from sumcnt
order by total DESC
limit 1
```
