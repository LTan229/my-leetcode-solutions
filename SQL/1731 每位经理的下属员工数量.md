```SQL
select up.employee_id, up.name, COUNT(*) as reports_count, ROUND(AVG(lo.age)) as average_age
from Employees as up join Employees as lo on up.employee_id = lo.reports_to
group by up.employee_id
having reports_count > 0
order by up.employee_id
```
