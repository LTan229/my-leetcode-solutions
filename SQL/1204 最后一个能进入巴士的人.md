```SQL
with sumtable as(
    select *, SUM(weight) over(ORDER BY turn) AS sums
    -- (
    -- select SUM(weight) from Queue as q where not q.turn > Queue.turn 
    -- ) as sums
    from Queue
)
select person_name from sumtable
where not sums > 1000
order by turn desc
limit 1
```
