```SQL
select Product.product_id, Product.product_name
from Product
join (select product_id, MIN(sale_date) as min_date, MAX(sale_date) as max_date from Sales group by product_id) as minmax
on Product.product_id = minmax.product_id
where not minmax.min_date < '2019-01-01' and not minmax.max_date > '2019-03-31 23:59:59'
```
