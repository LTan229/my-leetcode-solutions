```SQL
with maxdate as(
    select product_id, MAX(change_date) as change_date
    from Products where change_date < '2019-08-17'
    group by product_id
),
uniqprod as(
    select distinct product_id from Products
)

select uniqprod.product_id, (CASE when Products.new_price is null then 10 else Products.new_price END) as price
from Products join maxdate 
on Products.product_id = maxdate.product_id and Products.change_date = maxdate.change_date
right join uniqprod on Products.product_id = uniqprod.product_id 
```
