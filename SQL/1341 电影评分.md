```SQL
with usercnt as (
    select user_id, COUNT(*) as cnt
    from MovieRating
    group by user_id
),
movieavg as (
    select movie_id, AVG(rating) as rat 
    from MovieRating
    where created_at between '2020-02-01' and '2020-02-29 23:59:59'
    group by movie_id
)
select * from (
    select Users.name as results
    from usercnt join Users on usercnt.user_id = Users.user_id
    order by cnt DESC, Users.name
    limit 1
) as user_results
union all
select * from (
    select Movies.title as results
    from movieavg join Movies on movieavg.movie_id = Movies.movie_id
    order by rat DESC, Movies.title
    limit 1
) as movie_results
```
