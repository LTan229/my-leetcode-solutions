```SQL
select d.name as Department, 
e1.name as Employee,
salary as Salary
from Employee as e1 join Department as d on e1.departmentId = d.id
where 3 > (
    select COUNT(distinct salary) 
    from Employee as e2
    where e1.departmentId = e2.departmentId and e2.salary > e1.salary
)
```
