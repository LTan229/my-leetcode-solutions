```SQL
with twvalcnt as (
    select *, COUNT(*) OVER(partition by tiv_2015) as valcnt 
    from Insurance
),
twcnts as (
    select *, COUNT(*) OVER(partition by lat, lon) as citycnt
    from twvalcnt
)
select ROUND(SUM(tiv_2016),2) as tiv_2016
from twcnts
where valcnt > 1 and citycnt = 1
```
