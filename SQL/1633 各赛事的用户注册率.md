```SQL
# Write your MySQL query statement below
select contest_id, ROUND(contest_count * 100 / COUNT(DISTINCT user_id), 2) as percentage
from (select contest_id, COUNT(contest_id) as contest_count
from Register
group by contest_id) as cc
cross join Users
group by contest_id
order by contest_count desc, contest_id
```
