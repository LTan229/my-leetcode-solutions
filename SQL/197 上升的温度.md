```SQL
# Write your MySQL query statement below

select today.id as Id
from Weather as yesterday join Weather as today on DATE_ADD(yesterday.recordDate, INTERVAL 1 DAY) = today.recordDate 
where yesterday.temperature < today.temperature
;
```
