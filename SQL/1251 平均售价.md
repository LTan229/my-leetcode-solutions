```SQL
# Write your MySQL query statement below

select Prices.product_id, ROUND(IFNULL(SUM(price * units) / SUM(units),0),2) as average_price
from 
Prices left join UnitsSold on Prices.product_id = UnitsSold.product_id and (not purchase_date > end_date) and (not purchase_date < start_date)
group by Prices.product_id
```
