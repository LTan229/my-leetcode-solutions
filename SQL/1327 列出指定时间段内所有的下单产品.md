```SQL
with sumtable as (
    select product_name, 
    SUM(unit) as unit
from Orders join Products on Products.product_id = Orders.product_id
where order_date >= '2020-02-01' and order_date <= '2020-02-29'
group by Products.product_id
)
select *
from sumtable
where unit >= 100
```
