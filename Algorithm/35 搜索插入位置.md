```python
class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        len_num = len(nums)
        left = 0
        right = len_num - 1

        while right >= left:
            mid = left + (right - left) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] > target:
                right = mid - 1
            else:
                left = mid + 1

        if left < len_num:  
            if nums[left] > target:
                return left 
            else:
                return left + 1
        else:
            return len_num
```
