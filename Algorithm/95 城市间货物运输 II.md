# 负环检测

```python
from collections import deque

ncity, nroad = map(int, input().split())

edges = [[] for _ in range(ncity + 1)]
for _ in range(nroad):
    s, t, v = map(int, input().split())
    edges[s].append((t, v))

distances = [float('inf') for _ in range(ncity + 1)]
inq = [False for _ in range(ncity + 1)]
minjump = [float('inf')] * (ncity + 1)

q = deque([1])

distances[1] = 0
inq[1] = True
minjump[1] = 0

while q:
    node = q.popleft()
    inq[node] = False
    for neib, dist in edges[node]:
        if distances[neib] > distances[node] + dist:
            distances[neib] = distances[node] + dist
            minjump[neib] = minjump[node] + 1
            if minjump[neib] == ncity:
                print('circle')
                exit()
            if not inq[neib]: 
	            q.append(neib)
	            inq[neib] = True

if distances[-1] == float('inf'):
    print("unconnected")
else:
    print(distances[-1])
```
