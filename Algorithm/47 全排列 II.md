class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        self.temp = []
        self.result = []
        self.nums = nums
        self.lennums = len(nums)
        self.dfs([0] * self.lennums)
        return self.result
    
    def dfs(self, used):
        if sum(used) == self.lennums:
            self.result.append(self.temp[:])
            return
        
        usedval = []
        for i in range(self.lennums):
            if used[i]: 
                continue
            if self.nums[i] in usedval:
                continue
            self.temp.append(self.nums[i])
            used[i] = 1
            usedval.append(self.nums[i]) # it is certain to be new
            self.dfs(used)
            self.temp.pop()
            used[i] = 0
        
        return
