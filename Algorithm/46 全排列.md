```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        self.temp = []
        self.result = []
        self.nums = nums
        self.numcnt = len(nums)
        self.search([0] * self.numcnt)
        return self.result
    
    def search(self, used):
        if sum(used) == self.numcnt:
            self.result.append(self.temp[:])
            return None
        
        for i in range(self.numcnt):
            if used[i]:
                continue
            self.temp.append(self.nums[i])
            used[i] = 1
            self.search(used)
            self.temp.pop()
            used[i] = 0
```
