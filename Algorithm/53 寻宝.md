# Prim
## 自己写的prim
```python
nnode, nedge = map(int, input().split())

mindist = [float('inf') for _ in range(nnode + 1)]
adjlist = [[] for _ in range(nnode + 1)]

for _ in range(nedge):
    v1, v2, val = map(int, input().split())
    adjlist[v1].append((v2, val))
    adjlist[v2].append((v1, val))

cur = 1
sums = 0
while True:
    mindist[cur] = 0
    for neib, dist in adjlist[cur]:
        mindist[neib] = min(mindist[neib], dist)
    minval = float('inf')
    minidx = 0
    for i in range(1, nnode + 1):
        if mindist[i] > 0 and mindist[i] < minval:
            minval = mindist[i]
            minidx = i
    if minval == float('inf'):
        break
    else:
        cur = minidx
        sums += minval

print(sums)
```

- 用权重0标记已经访问的节点可能导致问题
- 使用最小堆来维护当前的mindist，这样获取最小元素只要o1就可以

## Heapq改进版

不修改mindist，只加入值，这样mindist会有重复的记录，比如node1和node3的距离是5,但node2和node3的距离是3.然后heap就会把3弹出来，如果之后5再弹出来就依据visited把它扔掉。
```python
import heapq

nnode, nedge = map(int, input().split())

adjlist = [[] for _ in range(nnode + 1)]

for _ in range(nedge):
    v1, v2, val = map(int, input().split())
    adjlist[v1].append((v2, val))
    adjlist[v2].append((v1, val))

pq = [(1, 0)] # neib, dist

visited = [False for _ in range(nnode + 1)] # use another array to store the state of each node
sums = 0
nodecnt = 0 # cnt of node in tree
while pq:
    cur, dist = heapq.heappop(pq)

    if visited[cur]:
        continue
    
    visited[cur] = True
    sums += dist
    nodecnt += 1

    if nodecnt == nnode:
        break
    
    for neib, dist in adjlist[cur]:
        if not visited[neib]:
            heapq.heappush(pq, (neib, dist))
    
print(sums)
```

# Kruskal

```python
import heapq

nnode, nedge = map(int, input().split())

edges = []

for _ in range(nedge):
    v1, v2, val = map(int, input().split())
    heapq.heappush(edges, (val, v1, v2))

parents = [i for i in range(nnode + 1)]

def find(u):
    if parents[u] == u:
        return u
    else:
        parents[u] = find(parents[u])
        return parents[u]

sums = 0
edgecnt = 0
while edges:
    val, v1, v2 = heapq.heappop(edges)

    # a join function here
    p1 = find(v1)
    p2 = find(v2)
    if p1 == p2:
        continue
    else:
        parents[p2] = p1
        sums += val
        edgecnt += 1
    
    # early quitting
    if edgecnt == nnode - 1:
        break

print(sums)

# 不是把v2的parent设为p1,是找到他们的根节点进行操作
```
