```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:

    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        self.result = None
        self.findCnt(root, p, q)
        return self.result

    
    def findCnt(self, root, p, q):
        if not root:
            return 0
        lcnt = self.findCnt(root.left, p, q)
        rcnt = self.findCnt(root.right, p, q)
        scnt = 1 if root == p or root == q else 0  
        tot = lcnt + rcnt + scnt
        if not self.result and tot == 2:
            self.result = root
        return tot
```
