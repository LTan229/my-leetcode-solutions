# collections
```python
from collections import Counter

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        scnt = Counter(s)
        tcnt = Counter(t)
        if scnt == tcnt:
            return True
        return False
        
```

# hashtable
```python
from collections import Counter

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        sbins = [0] * 26
        tbins = [0] * 26
        baseline = ord('a')
        for char in s:
            idx = ord(char) - baseline
            sbins[idx] = sbins[idx] + 1
        for char in t:
            idx = ord(char) - baseline
            tbins[idx] = tbins[idx] + 1
        for si, ti in zip(sbins, tbins):
            if si != ti:
                return False
            
        return True
```

ord(): char to unique code
chr(): unique code to char

from collections import Counter

class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        ssorted = sorted(s)
        tsorted = sorted(t)
        return ssorted == tsorted
