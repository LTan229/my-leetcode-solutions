```python
n, m = input().split(' ')
n, m = int(n), int(m)

city = [list(map(int, input().split(' '))) for i in range(n)]

cur_sum = 0
row_sum = []
for i in range(n):
    cur_sum += sum(city[i])
    row_sum.append(cur_sum)

cur_sum = 0
col_sum = []
for j in range(m):
    cur_sum += sum([city[i][j] for i in range(n)])
    col_sum.append(cur_sum)

min_diff = float('inf')

if n > 1:
    for i in range(n - 1):
        temp_diff = abs(row_sum[-1] - 2 * row_sum[i])
        min_diff = min(min_diff, temp_diff)
else:
    temp_diff = row_sum[0]
    min_diff = min(min_diff, temp_diff)

if m > 1:
    for i in range(m - 1):
        temp_diff = abs(col_sum[-1] - 2 * col_sum[i])
        min_diff = min(min_diff, temp_diff)
else:
    temp_diff = col_sum[0]
    min_diff = min(min_diff, temp_diff)

print(min_diff)

```
