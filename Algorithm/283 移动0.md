# 双指针
```python
class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        lens = len(nums)
        i, j = 0, 0
        while i < lens and j < lens:
            nums[j] = nums[i]
            i += 1
            j += 1 if nums[j] != 0 else 0
        for k in range(j, lens):
            nums[k] = 0
        
```
