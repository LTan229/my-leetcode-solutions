# 我的
```python
class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        for char in s:
            if char in ['(', '{', '[']:
                stack.append(char)
            elif char in [')', '}', ']']:
                if not stack:
                    return False
                prev = stack[-1]
                if (prev.startswith('(') and char.startswith(')')) or (prev.startswith('[') and char.startswith(']')) or (prev.startswith('{') and char.startswith('}')):
                    stack.pop()
                else:
                    return False
            else:
                return False
        if not stack:
            return True
        return False

```

# 题解

```python
class Solution:
    def isValid(self, s: str) -> bool:
        if len(s) % 2 == 1:
            return False
        
        pairs = {
            ")": "(",
            "]": "[",
            "}": "{",
        }
        stack = list()
        for ch in s:
            if ch in pairs:
                if not stack or stack[-1] != pairs[ch]:
                    return False
                stack.pop()
            else:
                stack.append(ch)
        
        return not stack
```
