from collections import Counter

class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        self.cntlist = sorted(Counter(candidates).items()) # ascending order comparing the first item first
        # sorted(cnt.items(), key=lambda item: item[1]) to sort with another item
        self.temp = []
        self.results = []
        self.dfs(0, target)
        return self.results
    
    def dfs(self, cur, target):
        if target == 0:
            self.results.append(self.temp[:]) # shallow copy
            return None
        if cur == len(self.cntlist) or target < self.cntlist[cur][0]: # 蕴含了要按数字从小到大排列。因为如果 freq[pos][0] 已经大于 rest，那么后面还没有递归到的数也都大于 rest，这就说明不可能再选择若干个和为 rest 的数放入列表了。此时，我们就可以直接回溯。
            # if searched to the tail
            # or current val is larger than target (negative target included)
            return None

        self.dfs(cur + 1, target)

        curval = self.cntlist[cur][0]
        maxx = min(target//curval, self.cntlist[cur][1])
        for i in range(1, maxx + 1):
            self.temp.append(curval)
            self.dfs(cur + 1, target - i * curval)
        
        for i in range(1, maxx + 1):
            self.temp.pop()
