```python
class Solution:
    def maxProfit(self, prices: List[int], fee: int) -> int:
        dp = [- prices[0], 0]
        for i in range(1, len(prices)):
            wval = max(dp[0], dp[1] - prices[i])
            woval = max(dp[1],dp[0] + prices[i] - fee)
            dp = [wval, woval]
        return max(dp)
```
