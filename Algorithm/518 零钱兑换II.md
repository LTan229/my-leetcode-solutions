```python
class Solution:
    def change(self, amount: int, coins: List[int]) -> int:
        # dp[j] 是考虑特定面额范围硬币的情况下凑出总额j的情况数
        dp = [0] * (amount + 1)
        dp[0] = 1
        for v in coins:
            for j in range(v, amount + 1):
                dp[j] = dp[j] + dp[j - v]
                # print(v, j, dp[j])
        return dp[-1]
```
