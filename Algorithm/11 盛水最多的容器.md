class Solution:
    def maxArea(self, height: List[int]) -> int:
        lens = len(height)
        i = 0
        j = lens - 1
        maxsize = -1
        while j > i:
            cursize = (j - i) * min(height[i], height[j])
            maxsize = max(maxsize, cursize)
            if height[i] < height[j]:
                i += 1
            else:
                j -= 1
        return maxsize

# first attempt: initiate i and j at both ends and move the pointer which produces bigger area (j - i - 1) * min(height[i + 1], height[j])
# second attempt: innit i and j at mid array and move the pointer which produces bigger area
# third attempt: maxheap? minheap? cant decide. useful for 3d case but do not eliminate the comparison even if all highest pillars are recorded
# the first attempt is close, but my approach searches for local optimal instead of global optimals. note that the ara is bound by the shorter pillar. as the width decreases, the area obtained by moving the higher pillar can't be larger than the curent area.
