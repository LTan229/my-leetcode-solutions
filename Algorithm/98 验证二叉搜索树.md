# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        if not root:
            return True
        if not root.left and not root.right:
            return True
        vleft, lmin, lmax = self.isValidRange(root.left, True)
        vright, rmin, rmax = self.isValidRange(root.right, False)
        vleft = vleft and (lmax is None or lmax < root.val)
        vright = vright and (rmin is None or rmin > root.val)
        return vleft and vright
    
    def isValidRange(self, root, isleft):
        if not root:
            return True, None, None

        if not root.left: # left node
            vleft = True
            lmin, lmax = root.val, root.val
        else:
            vleft, lmin, lmax = self.isValidRange(root.left, True)
            # print(vleft, lmin, lmax)
            vleft = vleft and root.val > lmax # left subtree is valid and smaller than root

        if not root.right:
            vright = True
            rmax, rmin = root.val, root.val
        else:
            vright, rmin, rmax = self.isValidRange(root.right, False)
            # print(vright, rmin, rmax)
            vright = vright and root.val < rmin # right subtree is valid and bigger than root

        return vleft and vright, min(lmin, rmin, root.val), max(lmax, rmax, root.val)

# 递归时向下传递子节点有效范围
```python
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        
        # 定义一个辅助函数
        # node: 当前节点
        # lower: 当前节点必须大于的值（下界）
        # upper: 当前节点必须小于的值（上界）
        def helper(node, lower, upper):
            if not node:
                # 空节点被认为是有效的
                return True
            
            # 1. 检查当前节点的值是否在有效范围内
            if not (lower < node.val < upper):
                return False
            
            # 2. 递归检查子树
            # 检查左子树：它的上界被收紧为 node.val
            # 检查右子树：它的下界被收紧为 node.val
            return (helper(node.left, lower, node.val) and
                    helper(node.right, node.val, upper))

        # 初始调用，根节点的范围是正无穷到负无穷
        return helper(root, float('-inf'), float('inf'))
```
# 中序遍历检验递增
## 迭代
```python
from collections import deque
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        stack = deque([])
        cur = root
        prev = None
        while stack or cur:
            if cur:
                stack.append(cur)
                cur = cur.left
            else:
                cur = stack.pop()
                if prev and prev.val >= cur.val:
                    return False
                prev = cur
                cur = cur.right
        return True
```
