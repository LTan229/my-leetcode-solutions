# 二维数组
```python
m, n = map(int, input().split())
w = list(map(int, input().split()))
v = list(map(int, input().split()))

dp = [[0] * (n + 1) for _ in range(m)]

# omitted initiation for first col
for j in range(w[0], n + 1): # limited modificaiton range 
    dp[0][j] = v[0] # omitted case judgement

for i in range(1, m): # obj id
    for j in range(1, n + 1): # bag volumn
        if j < w[i]:
            dp[i][j] = dp[i - 1][j] # omitted standalone var definition
        else:
            dp[i][j] = max(dp[i - 1][j], v[i] + dp[i - 1][j - w[i]])

print(dp[-1][-1])
```

# 一维数组

```python
m, n = map(int, input().split())
w = list(map(int, input().split()))
v = list(map(int, input().split()))

dp = [0] * (n + 1) 

# omitted initiation for first col
for j in range(w[0], n + 1): # limited modificaiton range 
    dp[j] = v[0] # omitted case judgement

for i in range(1, m): # obj id
    for j in range(n, 0, -1): # bag volumn
        if j < w[i]:
            dp[j] = dp[j] # omitted standalone var definition
        else:
            dp[j] = max(dp[j], v[i] + dp[j - w[i]])

print(dp[-1])
```
