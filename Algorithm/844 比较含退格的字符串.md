从后向前遇到退格则遍历这一组删除、执行足够数量的删除、如果都有效则比较。
```python
class Solution:
    def backspaceCompare(self, s: str, t: str) -> bool:
        s_idx = len(s) - 1
        t_idx = len(t) - 1
        skip_s, skip_t = 0, 0
        while s_idx >= 0 or t_idx >= 0:
            # find next valid position
            while s_idx >= 0:
                # skip_s = 0
                if s[s_idx] == '#':
                    skip_s += 1
                    s_idx -= 1
                elif skip_s > 0:
                    skip_s -= 1
                    s_idx -= 1
                else:
                    break
            while t_idx >= 0:
                # skip_t = 0
                if t[t_idx] == '#':
                    skip_t += 1
                    t_idx -= 1
                elif skip_t > 0:
                    skip_t -= 1
                    t_idx -= 1
                else:
                    break
            # check quit
            print(s_idx, t_idx)
            print(skip_s, skip_t)
            if s_idx >= 0 and t_idx >= 0: 
                if s[s_idx] != t[t_idx]:
                    print('hi')
                    return False
            elif s_idx >= 0 or t_idx >= 0: 
                return False
            
            # move pointer
            s_idx -= 1
            t_idx -= 1

        return True
```
