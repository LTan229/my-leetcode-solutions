# 第一想法

```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if not s:
            return 0
        i, j = 0, 1 # [i,j)
        d = {s[0]:0}
        maxlen = 1
        while j < len(s):
            if s[j] in d:
                dstidx = d[s[j]] + 1
                while i < dstidx:
                    d.pop(s[i])
                    i += 1
            d[s[j]] = j
            j += 1
            maxlen = max(maxlen, j - i)
        return maxlen
```

# 改进

```python
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        d = {}
        i = 0
        maxlen = 0
        for j in range(len(s)): # [i,j]
            if s[j] in d:
                # 发现重复字符时，如果重复字符比i大（在窗口里），则i跳到重复字符最后一次出现右边，否则，重复字符并不在当前窗口中，相当于没有重复字符，不进行操作。
                i = max(i, d[s[j]] + 1)
            d[s[j]] = j
            maxlen = max(maxlen, j - i + 1)
        return maxlen
```
