class Solution:
    def findSubsequences(self, nums: List[int]) -> List[List[int]]:
        self.results = []
        self.temp = []
        self.nums = nums
        self.lens = len(nums)
        self.dfs(0)
        return self.results
    
    def dfs(self, start):
        if start == self.lens:
            return
        
        used = []
        for idx in range(start, self.lens):
            curnum = self.nums[idx]
            if curnum in used:
                continue
            if not self.temp:
                self.temp.append(curnum)
                self.dfs(idx + 1)
                self.temp.pop()
                used.append(curnum)
            elif curnum >= self.temp[-1]: # at least 1 item in temp
                self.temp.append(curnum) # temp has at least two item
                self.results.append(self.temp[:]) 
                self.dfs(idx + 1)
                self.temp.pop()
                used.append(curnum)
        return
