```python
class Solution:
    def rob(self, nums: List[int]) -> int:
        dp = [0, nums[0]]
        for i in range(1, len(nums)):
            # print(i, nums[i])
            nextval = max(dp[i], dp[i - 1] + nums[i])
            dp.append(nextval)
        return dp[-1]
```
