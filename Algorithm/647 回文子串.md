```python
class Solution:
    def countSubstrings(self, s: str) -> int:
        ls = len(s)
        dp = [[0] * (ls + 1) for _ in range(ls + 1)]
        summ = 0
        for i in range(1, ls + 1):
            for j in range(i, 0, -1):
                if s[i - 1] == s[j - 1]:
                    if i - j < 2:
                        dp[i][j] = 1
                    else:
                        dp[i][j] = dp[i - 1][j + 1]
                summ += dp[i][j]
        return summ
```
