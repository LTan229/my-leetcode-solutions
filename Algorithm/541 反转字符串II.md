class Solution:
    def reverseStr(self, s: str, k: int) -> str:
        if k == 1:
            return s
        lens = len(s)
        result = ""
        for left in range(0, lens, 2 * k):
            mid = min(lens, left + k)
            right = min(lens, mid + k)
            result += s[left : mid][::-1]
            result += s[mid : right]      
        return result
